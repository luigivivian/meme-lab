```typescript
interface AuthContextType {
  user: { email: string; role: string } | null;
  isAuthenticated: boolean;
  isLoading: boolean;
  login: (email: string, password: string) => Promise<void>;
  register: (email: string, password: string) => Promise<void>;

```

**Default context value:** user: null, isAuthenticated: false, isLoading: true, login/register/logout: async no-ops.

**localStorage keys (per D-01):** `access_token` and `refresh_token`.

**AuthProvider component implementation:**

1. **State:** `user` (object|null), `isLoading` (boolean, starts true).
2. **Computed:** `isAuthenticated = user !== null`.
3. **Hydration (useEffect on mount):**
   - Read `access_token` from localStorage.
   - If token exists, call `fetch("/api/auth/me", { headers: { Authorization: "Bearer " + token } })`.
   - If 200, parse JSON and set `user` to `{ email: data.email, role: data.role }`.
   - If 401 or error, call `localStorage.removeItem("access_token")` and `localStorage.removeItem("refresh_token")`, set `user` to null.
   - Set `isLoading` to false in all cases (finally block).
   - Use a direct fetch() call here (not the api.ts request() function) to avoid circular dependency during hydration — api.ts will handle 401 redirect but we don't want redirect during initial hydration check.

4. **login(email, password) — useCallback:**
   - `const res = await fetch("/api/auth/login", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ email, password }) })`.
   - If `!res.ok`, parse body: `const body = await res.json().catch(() => null)`. Throw `new Error(body?.detail || "Erro no servidor. Tente novamente em alguns instantes.")`.
   - If 200, parse JSON: `{ access_token, refresh_token }`.
   - `localStorage.setItem("access_token", access_token)`, `localStorage.setItem("refresh_token", refresh_token)`.
   - Fetch `/api/auth/me` with the new token to get user data, set `user`.
   - `window.location.href = "/dashboard"` (per D-09 — redirect to dashboard after login).

5. **register(email, password) — useCallback:**
   - `const res = await fetch("/api/auth/register", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ email, password }) })`.
   - If `!res.ok`, parse body: `const body = await res.json().catch(() => null)`. Throw `new Error(body?.detail || "Erro no servidor. Tente novamente em alguns instantes.")`.
   - If 201, call `await login(email, password)` to auto-login (per D-10).

6. **logout() — useCallback:**
   - Read `refresh_token` from localStorage.
   - If refresh_token exists, call `fetch("/api/auth/logout", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ refresh_token }) })`. Ignore errors (best-effort server-side invalidation).
   - `localStorage.removeItem("access_token")`, `localStorage.removeItem("refresh_token")`.
   - Set `user` to null.
   - `window.location.href = "/login"` (per D-11).

**Export:** `AuthProvider` component and `useAuth` hook (using useContext).

**Modify `memelab/src/lib/api.ts`:**

1. In the `request()` function, BEFORE the existing fetch call, add token injection (per D-08 — reads from localStorage directly, not React context):

```typescript
async function request<T>(
  path: string,
  options?: RequestInit
): Promise<T> {
  const token = typeof window !== "undefined"
    ? localStorage.getItem("access_token")
    : null;

  const headers: Record<string, string> = {
    "Content-Type": "application/json",
    ...((options?.headers as Record<string, string>) || {}),
  };

  if (token) {
    headers["Authorization"] = `Bearer ${token}`;
  }

  const res = await fetch(`${BASE}${path}`, {
    ...options,
    headers,
  });

  if (!res.ok) {
    // On 401, redirect to login (per D-02)
    if (res.status === 401 && typeof window !== "undefined") {
      localStorage.removeItem("access_token");
      localStorage.removeItem("refresh_token");
      window.location.href = "/login";
      throw new Error("Sessao expirada");
    }
    const text = await res.text().catch(() => res.statusText);
    throw new Error(`API ${res.status}: ${text}`);
  }
  return res.json();
}
```

Key changes from existing code:
- Add `token` variable reading from localStorage with SSR guard (`typeof window !== "undefined"`)
- Build headers object explicitly (allows spreading user headers AND adding Authorization)
- Add 401 handler that clears tokens and redirects to /login
- Keep all existing types and API functions unchanged

```typescript
};

export default function RootLayout({ children }: { children: React.ReactNode }) {

        </AuthProvider>
      </body>
    </html>

}
```

Note: `metadata` export works in server components that import client components as children — this is standard Next.js pattern. The AuthProvider is a client component imported into a server layout.
