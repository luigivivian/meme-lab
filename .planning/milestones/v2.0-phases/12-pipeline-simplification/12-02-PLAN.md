    ```typescript
    // --- Manual Pipeline (Phase 12) ---
    export const manualRun = (params: ManualRunParams) =>
      request<PipelineRunResult>("/pipeline/manual-run", {
        method: "POST",
        body: JSON.stringify(params),
      });

    export const approveContent = (packageId: number) =>
      request<ApprovalResponse>(`/pipeline/content/${packageId}/approve`, { method: "PATCH" });

    export const rejectContent = (packageId: number) =>
      request<ApprovalResponse>(`/pipeline/content/${packageId}/reject`, { method: "PATCH" });

    export const unrejectContent = (packageId: number) =>
      request<ApprovalResponse>(`/pipeline/content/${packageId}/unreject`, { method: "PATCH" });

    export const bulkApproveContent = (packageIds: number[]) =>
      request<ApprovalResponse>("/pipeline/content/bulk-approve", {
        method: "PATCH",
        body: JSON.stringify({ package_ids: packageIds }),
      });

    export const bulkRejectContent = (packageIds: number[]) =>
      request<ApprovalResponse>("/pipeline/content/bulk-reject", {
        method: "PATCH",
        body: JSON.stringify({ package_ids: packageIds }),
      });

    export const uploadBackground = async (file: File, characterSlug: string) => {
      const token = typeof window !== "undefined" ? localStorage.getItem("access_token") : null;
      const formData = new FormData();
      formData.append("file", file);
      const headers: Record<string, string> = {};
      if (token) headers["Authorization"] = `Bearer ${token}`;
      const res = await fetch(
        `${BASE}/pipeline/backgrounds/upload?character_slug=${encodeURIComponent(characterSlug)}`,
        { method: "POST", headers, body: formData }
      );
      if (!res.ok) {
        const text = await res.text().catch(() => res.statusText);
        throw new Error(`Upload ${res.status}: ${text}`);
      }
      return res.json() as Promise<{ filename: string; path: string; width: number; height: number }>;
    };

    export const listBackgrounds = (characterSlug: string) =>
      request<{ backgrounds: BackgroundFile[] }>(`/pipeline/backgrounds/${encodeURIComponent(characterSlug)}`);

    export const getThemesWithColors = () =>
      request<{ themes: ThemeWithColors[] }>("/pipeline/themes");
    ```
    Note: `uploadBackground` uses `FormData` (not JSON Content-Type), so it bypasses the standard `request()` helper and calls `fetch` directly — do NOT set `Content-Type: application/json` header for multipart uploads.

    **2. Create useManualPipeline hook in memelab/src/hooks/use-pipeline.ts:**

    Add a new `useManualPipeline` hook below the existing `usePipeline` export. This hook:
    - Has state: `{ isRunning, runId, results: ContentPackage[], error, progress }`
    - `run(params: ManualRunParams)`: calls `api.manualRun(params)`, gets run_id, polls `api.getPipelineStatus(runId)` every 2 seconds until completed/error
    - On completion: extracts `content` array from PipelineRunResult, sets as `results`
    - `approve(packageId)`: calls `api.approveContent(packageId)`, updates local results array optimistically (set approval_status="approved")
    - `reject(packageId)`: calls `api.rejectContent(packageId)`, updates local results optimistically (set approval_status="rejected")
    - `unreject(packageId)`: calls `api.unrejectContent(packageId)`, updates local results optimistically (set approval_status="pending")
    - `bulkApprove()`: calls `api.bulkApproveContent(results.map(r => r.id))`, updates all to "approved"
    - `bulkReject()`: calls `api.bulkRejectContent(results.map(r => r.id))`, updates all to "rejected"
    - `reset()`: clears all state

    The results array needs `id` for approve/reject. The PipelineRunResult.content is of type ContentPackage which may not have `id`. The manual-run endpoint should return packages with their DB `id`. Add an `id?: number` field to the ContentPackage interface if not present. The manual-run response must include the package DB IDs — ensure the serializer in Plan 01 includes `id`.

    Pattern: follow the existing `usePipeline` hook structure (useState + useCallback + useRef for polling interval + useEffect for cleanup).
