Replace rigid keyword-based theme mapping with LLM intelligence, increase curator throughput, and add relevance filtering.

The relevance filter prevents irrelevant trends from wasting generation resources.

    ```python
    import json
    from src.llm_client import generate

    # In-memory cache for LLM theme mappings (topic -> situacao_key)
    _theme_cache: dict[str, str] = {}

    _THEME_MAPPING_PROMPT = """You are a visual theme selector for a Brazilian meme generator.
    Given a trending topic, select the single best visual theme from this list:

    {theme_list}

    Rules:
    - Return ONLY the theme key (e.g., "cafe"), nothing else
    - Match the topic's mood and context to the theme's visual scene
    - When in doubt, prefer "cotidiano" or "generico" over a bad match
    """

    def _llm_map_theme(topic: str, humor_angle: str, all_keys: list[str]) -> str | None:
        """Map a topic to the best situacao_key via LLM. Returns None on failure."""
        cache_key = f"{topic}|{humor_angle}".lower().strip()
        if cache_key in _theme_cache:
            return _theme_cache[cache_key]

        theme_descriptions = {
            "sabedoria": "wise contemplation, reading ancient books",
            "confusao": "bewildered, confused, scratching head",
            "segunda_feira": "tired monday morning, sleepy with coffee",
            "vitoria": "triumphant celebration, staff raised high",
            "tecnologia": "holding smartphone/device, digital runes",
            "cafe": "holding coffee mug, content expression",
            "comida": "cooking cauldron, floating ingredients",
            "trabalho": "at desk with laptop, focused/stressed",
            "relaxando": "sleeping in chair, hat over eyes",
            "meditando": "cross-legged meditation, serene aura",
            "relacionamento": "holding heart crystal, romantic mood",
            "confronto": "stern expression, staff with energy, dramatic",
            "surpresa": "wide eyes, shocked expression, hands raised",
            "cotidiano": "relaxed casual pose, holding ale mug",
            "descanso": "peaceful sleeping, chair, floating particles",
            "internet": "looking at crystal ball like screen, amused",
            "generico": "standing with staff, wise serious expression",
        }

        theme_list = "\n".join(
            f"- {key}: {theme_descriptions.get(key, key)}"
            for key in all_keys
        )

        try:
            result = generate(
                system_prompt=_THEME_MAPPING_PROMPT.format(theme_list=theme_list),
                user_message=f"Topic: {topic}\nHumor angle: {humor_angle}\n\nBest theme key:",
                max_tokens=20,
                tier="lite",
            ).strip().lower().replace('"', '').replace("'", "")

            # Validate result is a known key
            if result in all_keys:
                _theme_cache[cache_key] = result
                return result
            else:
                logger.warning(f"LLM returned unknown theme '{result}' for '{topic}', falling back")
                return None
        except Exception as e:
            logger.warning(f"LLM theme mapping failed for '{topic}': {e}")
            return None
    ```

    2. Replace `_match_situacao_diversa()` to use LLM mapping first, KEYWORD_MAP as fallback, random as last resort:

    ```python
    def _match_situacao_diversa(self, topic, humor_angle, used_keys, shuffled_pool) -> str:
        """Map topic to situacao_key: LLM first, keyword fallback, random last."""
        # Try LLM mapping first (per D-12)
        llm_key = _llm_map_theme(topic, humor_angle, self._all_keys)
        if llm_key and llm_key not in used_keys:
            logger.debug(f"  Situacao via LLM: '{topic}' -> {llm_key}")
            return llm_key

        # Fallback: keyword match (kept for offline/error resilience)
        combined = f"{topic} {humor_angle}".lower()
        for keyword, situacao_key in KEYWORD_MAP.items():
            if keyword in combined and situacao_key not in used_keys:
                logger.debug(f"  Situacao via keyword fallback: '{keyword}' -> {situacao_key}")
                return situacao_key

        # Last resort: random from pool
        for key in shuffled_pool:
            if key not in used_keys:
                logger.debug(f"  Situacao random (no LLM/keyword match): {key}")
                return key

        key = random.choice(self._all_keys)
        logger.debug(f"  Situacao random (pool exhausted): {key}")
        return key
    ```
