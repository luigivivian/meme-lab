    ```python
    import re

    # Common English words that are OK in Brazilian Portuguese memes (Brazilianisms)
    _ALLOWED_ENGLISH = {
        "fake", "hype", "crush", "stalker", "like", "selfie", "stories",
        "delivery", "home office", "wifi", "online", "streaming",
        "playlist", "lo-fi", "lofi", "spoiler", "meme", "trend",
        "reels", "feed", "post", "hack", "top", "vibe", "mood",
        "cringe", "ghosting", "match", "date", "ok", "sorry",
    }

    # Regex to find English-looking words (3+ letters, all ASCII, not common PT)
    _ENGLISH_WORD_RE = re.compile(r'\b[a-zA-Z]{3,}\b')

    # Common Portuguese words to verify language
    _PT_MARKERS = {
        "que", "nao", "com", "uma", "pra", "quando", "mas", "meu",
        "minha", "voce", "essa", "esse", "aqui", "mais", "como",
        "vai", "vou", "sou", "tem", "esta", "seu", "sua",
    }

    def _validate_phrase(phrase: str, max_chars: int = 120) -> tuple[bool, str]:
        """Validate a phrase for image overlay suitability.

        Returns (is_valid, reason). Reason is empty if valid.
        """
        # Check 1: Length
        if len(phrase) > max_chars:
            return False, f"too_long ({len(phrase)} > {max_chars})"

        if len(phrase) < 10:
            return False, f"too_short ({len(phrase)} chars)"

        # Check 2: Language — ensure it's predominantly Portuguese
        words = phrase.lower().split()
        if len(words) < 3:
            return False, "too_few_words"

        # Count words that look English and aren't in the allowed list
        english_count = 0
        for word in words:
            clean_word = re.sub(r'[^a-zA-Z]', '', word)
            if (clean_word and len(clean_word) >= 3
                    and clean_word.lower() not in _ALLOWED_ENGLISH
                    and clean_word.lower() not in _PT_MARKERS
                    and clean_word.isascii()):
                # Check if it could be Portuguese (has common PT suffixes)
                pt_suffixes = ("ção", "mente", "ando", "endo", "indo", "ado", "ido", "oso", "osa")
                if not any(word.endswith(s) for s in pt_suffixes):
                    english_count += 1

        english_ratio = english_count / len(words) if words else 0
        if english_ratio > 0.4:
            return False, f"too_much_english ({english_count}/{len(words)} words)"

        # Check 3: Not empty/placeholder
        if phrase.strip().startswith("[") or phrase.strip().startswith("{"):
            return False, "placeholder_text"

        return True, ""
    ```

    from src.llm_client import generate as llm_generate

    def _check_coherence(topic: str, humor_angle: str, situacao_key: str) -> bool:
        """Check if the visual theme matches the phrase topic.

        Uses a quick LLM check: 'Does {situacao_key} theme make sense for {topic}?'
        Returns True if coherent, False if mismatch detected.
        """
        # Known good matches (skip LLM call for obvious ones)
        obvious_matches = {
            "cafe": ["cafe", "coffee", "cafeina"],
            "trabalho": ["trabalho", "emprego", "chefe", "escritorio", "home office"],
            "segunda_feira": ["segunda", "monday", "começo de semana"],
            "tecnologia": ["wifi", "internet", "celular", "app", "rede social"],
            "comida": ["comida", "cozinha", "fome", "almoco", "jantar", "miojo"],
            "relacionamento": ["namoro", "crush", "ex", "casal", "amor"],
            "descanso": ["preguica", "sono", "dormir", "ferias", "feriado"],
            "relaxando": ["relaxar", "descanso", "tranquilo", "paz"],
        }

        combined = f"{topic} {humor_angle}".lower()
        if situacao_key in obvious_matches:
            for kw in obvious_matches[situacao_key]:
                if kw in combined:
                    return True  # Obvious match, skip LLM

        # For non-obvious cases, use LLM quick check
        try:
            result = llm_generate(
                system_prompt="Answer only YES or NO.",
                user_message=(
                    f"Does the visual theme '{situacao_key}' make sense for a meme about: '{topic}'?\n"
                    f"Humor angle: '{humor_angle}'\n"
                    f"Answer YES if the theme fits the topic, NO if there's a disconnect."
                ),
                max_tokens=5,
                tier="lite",
            ).strip().upper()
            return result.startswith("YES") or result.startswith("SIM")
        except Exception:
            return True  # On error, assume coherent (don't block generation)
    ```

    Integrate the coherence check into the generation flow. Find where `work_order.situacao_key` is used to select the image theme/background. Before image generation:

                work_order.situacao_key = new_key
        except Exception as e:
            logger.warning(f"[{work_order.order_id}] Remap failed: {e}, keeping original theme")
    ```
