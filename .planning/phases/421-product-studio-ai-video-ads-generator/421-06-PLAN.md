    2. Create `memelab/src/hooks/use-ads.ts`:
    ```typescript
    "use client";
    import useSWR from "swr";
    import type { AdJob } from "@/lib/api";

    const fetcher = (url: string) => fetch(url).then(r => { if (!r.ok) throw new Error(); return r.json(); });

    export function useAdJobs() {
      return useSWR<AdJob[]>("/api/ads/jobs", fetcher, { refreshInterval: 5000 });
    }

    export function useAdJob(jobId: string | null) {
      return useSWR<AdJob>(jobId ? `/api/ads/${jobId}` : null, fetcher, { refreshInterval: 3000 });
    }

    export function useAdSteps(jobId: string | null) {
      return useSWR(jobId ? `/api/ads/${jobId}/steps` : null, fetcher, { refreshInterval: 2000 });
    }
    ```
